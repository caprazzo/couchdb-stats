1 /**
  2 	Takes multple value objects
  3 	<pre>[{time:1200, requests:500}, {time:55, requests:2000}]</pre>
  4 	
  5 	sums times and requests and calcs the average
  6 	
  7 	<pre>{time:1255, requests:2500, avg: 0.502}</pre>
  8 
  9 	@name views.request_time.reduce
 10 	
 11 */
 12 function(keys, values, rereduce) {
 13 	var rt = {time:0, requests: 0};
 14 	values.forEach(function(value) {
 15 		rt.time += value.time;
 16 		rt.requests += value.requests;
 17 	});
 18 	rt.avg = rt.time / rt.requests;
 19 	return rt;
 20 }
