1 /**
  2 	Given a view where the key is a 6-level date and value an integer,
  3 	renders an html table conformant to the timepedia chronoscope microformat
  4 	(http://timepedia.org/chronoscope/docs/microformats/).
  5 	<br/><br/>
  6 	MANDATORY group_level 1-6 to control grouping<br/>
  7 	OPTIONAL table_only [true|false] to disable/enable chronoscope (default enabled)
  8 	
  9 	@name views.table
 10 */
 11 function(head, req) {
 12 	// !json templates
 13 	// !code lib/mustache.js
 14 	// !code lib/stats.js
 15 	
 16  	start({"headers":{"Content-Type" : "text/html"}});
 17 	
 18 	var dt = get_dtformat(parseInt(req.query.group_level));
 19 	var table_only = (req.query.table_only && req.query.table_only=='true');
 20 	
 21 	if (!table_only) {
 22 		send(Mustache.to_html(templates.app_head, {root:app_root(req.path)}));
 23 	}
 24 
 25 	send(Mustache.to_html(templates.chronoscope.table_head, {dtformat:dt.dtformat}));
 26 	while(row = getRow()) {		
 27 		send(Mustache.to_html(templates.chronoscope.table_item, {
 28 			key: dt.f_key(row.key),
 29 			value: row.value
 30 		}));
 31 		send('\n');
 32 	}
 33 	send(Mustache.to_html(templates.chronoscope.table_foot));
 34 
 35 	if (!table_only) {
 36 		send(Mustache.to_html(templates.script, {url:'http://api.timepedia.org/widget/'}));
 37 		send(Mustache.to_html(templates.stylesheet, {href: 'http://api.timepedia.org/widget/Chronoscope.css'}));
 38 		send(Mustache.to_html(templates.app_foot));
 39 	}
 40 
 41 }
