1 /**
  2 	Takes multiple value objects <pre>{ POST: 1, GET: 2, ...}</pre>
  3 	amd sums the properties together.
  4 
  5 	@name views.http_requests_hits_by_method.reduce
  6 */
  7 function(keys, values, rereduce) {
  8 	var ret = {};
  9 	values.forEach(function(value) {
 10 		for (var METHOD in value) {
 11 			if (!ret[METHOD]) ret[METHOD] = 0;
 12 			ret[METHOD] += value[METHOD];
 13 		}
 14 	});
 15 	return ret;
 16 }
