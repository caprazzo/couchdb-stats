1 /**
  2 	Generates a UTC dates array suitable for emission as 6-level key in a map function.
  3 	The array is in the form [year, month, day, hour, 5-minutes-period, minute]
  4 	
  5 	5-minute-period is useful to group by 5-minutes:
  6 	minute 0,1,2,3,4 --> period 0
  7 	minute 5,6,7,8,9 --> period 1
  8 	...
  9 	
 10 	group_level=6 and group_level=5 will only be different if
 11 	stats are collected each minute
 12 		
 13 	@name stats.extended_time_key_from_json_date
 14  */
 15 function extended_time_key_from_json_date(str_timestamp) {	
 16 	var date = new Date(str_timestamp);
 17 	return [
 18 		date.getUTCFullYear(),
 19 		date.getUTCMonth() + 1,
 20 		date.getUTCDate(),
 21 		date.getUTCHours(),
 22 		// 5 minutes step
 23 		Math.floor(date.getUTCMinutes() / 5),
 24 		date.getUTCMinutes()
 25 	];
 26 }

 30 		return {
 31 			f_key: function(key) {
 32 				return [ key[0], key[1], key[2], key[3], key[5] ].join('-')
 33 			},
 34 			dtformat: 'yyyy-M-d-H-m'			
 35 		}
 36 	}
 37 	else if (group_level == 5) {
 38 		return {
 39 			f_key: function(key) {
 40 				return [ key[0], key[1], key[2], key[3], key[4]*5 ].join('-')
 41 			},
 42 			dtformat: 'yyyy-M-d-H-m'			
 43 		}
 44 	}
 45 	else if (group_level < 5) {
 46 		return {
 47 			f_key: function(key) {
 48 				return key.join('-');
 49 			},
 50 			dtformat: 'yyyy-M-d-H-m'.split('-').slice(0,parseInt(req.query.group_level)).join('-')
