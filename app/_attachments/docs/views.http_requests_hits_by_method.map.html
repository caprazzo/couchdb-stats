1 /**
  2 	Emits a 6-level extended time key and an object that counts
  3 	httpd hits for each method
  4 	<pre>emit([2010, 1, 1, 20, 30, 0, 4], { POST: 1, GET: 2, ...})</pre>
  5 	
  6 	@name views.http_requests_hits_by_method.map
  7 */
  8 function(doc) {
  9 	// !code lib/stats.js
 10 	if (!doc.timestamp) return;
 11 	var key = extended_time_key_from_json_date(doc.timestamp.replace(/-/g,'/'));
 12 	var ret = {};
 13 	for(var METHOD in doc.httpd_request_methods) {
 14 		var sum = doc.httpd_request_methods[METHOD].sum;		
 15 		ret[METHOD] = ((sum) ? sum : 0);		
 16 	}
 17 	emit(key, ret);
 18 }
